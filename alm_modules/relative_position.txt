module relative_position
    depends on root
    sort declarations
        relative_move :: actions
            attributes
                actor : agents
                origin : points
                dest : points


    function declarations
        fluent
            basic
                is_near : agents * agents -> booleans
                is_above : agents * agents -> booleans
                is_below : agents * agents -> booleans
                is_near : points * points -> booleans
                is_above : points * points -> booleans
                is_below : points * points -> booleans

    axioms
        is_near(X,Y) if -is_near(X,Y).
        -is_near(X,Y) if is_near(X,Y).
        is_near(X,Y) if is_near(Y,X).
        -is_near(Y,X) if -is_near(X,Y).

        -is_above(X,Y) if is_above(Y,X).
        -is_below(X,Y) if is_below(Y,X).
        is_below(X,Y) if is_above(Y,X).
        is_above(X,Y) if is_below(Y,X).

        occurs(X) causes is_near(A,B) if instance(X,relative_move),
                                        actor(X) = A,
                                        dest(X) = C,
                                        at(B) = D,
                                        is_near(C,D).

        occurs(X) causes is_near(A,B) if instance(X,relative_move),
                                        actor(X) = B,
                                        dest(X) = C,
                                        at(A) = D,
                                        is_near(C,D).

        occurs(X) causes -is_near(A,B) if instance(X,relative_move),
                                        actor(X) = B,
                                        at(A) = C,
                                        dest(X) = D,
                                        -is_near(C,D).


        occurs(X) causes -is_near(B,A) if instance(X, relative_move),
                                        actor(X) = A,
                                        at(B) = C,
                                        dest(X) = D,
                                        -is_near(C,D).

        occurs(X) causes is_above(A,B) if instance(X, relative_move),
                                        actor(X) = A,
                                        dest(X) = C,
                                        at(B) = D,
                                        is_above(C,D).
                                        

        occurs(X) causes is_above(A,B) if instance(X, relative_move),
                                        actor(X) = B,
                                        dest(X) = D,
                                        at(A) = C,
                                        is_above(C,D).
